102void AnimatableLcd::printPretty(String text) //handelt zeilenumbrüche und schreibt zentriert
103{
104 clear();
105 int length = text.length();
106 if (length <= 16) {
107 printCentered(text, length);
108 return 0;
109 }
110 int spacePos = -1;
111 for (int i = 15; i >= 0; i--) {
112 if (text[i] == ' ') {
113 spacePos = i;
114 break;
115 }
116 }
117 String row1, row2;
118 if (spacePos != -1) {
119 row1 = text.substring(0, spacePos);
120 row2 = text.substring(spacePos + 1);
121 } else {
122 row1 = text.substring(0, 16);
123 row2 = text.substring(16);
124 }
125 printCentered(row1, row1.length(), 0);
126 printCentered(row2, row2.length(), 1);
127}
