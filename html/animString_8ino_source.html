82void LcdLoadingAnim::update()
83{
84 time_t time = millis();
85 short percent = (time - animStart) * 100 / duration;
86 if (time - lastRefresh > stepDuration) {
87 short nToFill = percent * 9 / 100;
88 if (nToFill == 0) {
89 return;
90 }

92 lcd->write(1);

94 }
95 lcd->setCursor(8 + LOADING_BAR_OFFSET, 1);

113 time_t time = millis();

115 return;

117 lastRefresh = time;
