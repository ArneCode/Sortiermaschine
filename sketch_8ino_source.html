293void loop()
294{
295 callHandler.update();
296 lcd.update();
297 servo.updatePos();
298 stopButton.update();
299 if (doFlicker) {
300 unsigned short b = (millis() / 2) % 513; //helligkeit: 0-512
301 if (b > 255) {
302 b = 511 - b; //wenn b größer als 255, wird die helligkeit kleiner, Werte über 255 werden also "gespiegelt"
303 }
304 setLedColor(b, b / 2, 0); //orange
305 if (random(3) == 0) {
306 lcd.noBacklight();
307 } else {
308 lcd.backlight();
309 }
310 }
311 if (callHandler.running) {
312 return;
313 }
314 Farbe farbe = mesureColor();
315 servo.setSpeed(SERVO_SPEED_DEFAULT);
316 switch (farbe) {
317 case WHITE:
318 {
319 nWhite++;
320 Serial.println("white");
321 legBallAb<ANGLE_LEFT_HOLE>("Wei\342");
322 setLedColor(255, 255, 255);
323 break;
324 }
325 case BLACK:
326 {
327 nBlack++;
328 Serial.println("black");
329 legBallAb<ANGLE_RIGHT_HOLE>("Schwarz");
330 setLedColor(0, 0, 255);
331 break;
332 }
333 case NOTHING:
334 {
335 Serial.println("nothing");
336 callHandler.deleteCalls();
337 /*static*/ auto callsNothing = new Callable*[1] {
338 new LcdString(String("Ball einlegen W:") + nWhite + String(" S:") + nBlack + String(" O:") + nOrange, &lcd, 1000)
339 };
340 callHandler.setCalls(callsNothing, 1);
341 setLedColor(0, 255, 0);
342 break;
343 }
344 case ORANGE:
345 {
346 doFlicker = true;
347 nOrange++;
348 Serial.println("orange");
349 callHandler.deleteCalls();
350 /*static*/ auto callsOrange = new Callable*[10] {
351 new LcdString("Oran\2er B\2ll",&lcd,1000),
352 new LcdString("\2\2\2\2O\2\2OR\2\2R\2\2A\2\2N\2\2G\2\2\2E\2\2\2\2!\2\2",&lcd,0),
353 new FuncCall([]() {
354 servo.write(ANGLE_RIGHT_HOLE);

357 new FuncCall([]() {
358 servo.setSpeed(SERVO_SPEED_FAST);
